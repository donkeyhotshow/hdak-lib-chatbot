// dropdown toggle
function tog(name) {
  const dd = document.getElementById('dd-' + name);
  const btn = document.getElementById('btn-' + name);
  const ov = document.getElementById('ov');
  const open = dd.classList.contains('open');
  document.querySelectorAll('.dd').forEach(d => d.classList.remove('open'));
  document.querySelectorAll('.tb-btn').forEach(b => b.classList.remove('active'));
  ov.classList.remove('on');
  if (!open) { dd.classList.add('open'); btn.classList.add('active'); ov.classList.add('on'); }
}
document.getElementById('ov').addEventListener('click', () => {
  document.querySelectorAll('.dd').forEach(d => d.classList.remove('open'));
  document.querySelectorAll('.tb-btn').forEach(b => b.classList.remove('active'));
  document.getElementById('ov').classList.remove('on');
});

// lang
function setLang(el, code) {
  document.querySelectorAll('.lang-row').forEach(r => r.classList.remove('cur'));
  el.classList.add('cur');
  document.getElementById('btn-lang').innerHTML = '🌐 ' + code;
  tog('lang');
}

// textarea
const ta = document.getElementById('ta');
const sendBtn = document.getElementById('send');
ta.addEventListener('input', () => {
  ta.style.height = 'auto';
  ta.style.height = Math.min(ta.scrollHeight, 100) + 'px';
  sendBtn.disabled = !ta.value.trim();
});
ta.addEventListener('keydown', e => {
  if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); if (!sendBtn.disabled) sendMsg(); }
});
function sendMsg() { showView('chat'); ta.value = ''; sendBtn.disabled = true; ta.style.height = 'auto'; }

// chips
function useChip(el) {
  ta.value = el.textContent.replace(/^[^\s]+\s/, '');
  sendBtn.disabled = false;
  showView('chat');
  ta.focus();
}

// view switcher
function showView(v) {
  document.getElementById('view-empty').style.display = v==='empty' ? 'flex' : 'none';
  document.getElementById('view-chat').style.display  = v==='chat'  ? 'flex' : 'none';
  document.getElementById('sw-e').classList.toggle('on', v==='empty');
  document.getElementById('sw-c').classList.toggle('on', v==='chat');
}

// scroll chat to bottom
const chat = document.getElementById('view-chat');
if (chat) chat.scrollTop = chat.scrollHeight;
