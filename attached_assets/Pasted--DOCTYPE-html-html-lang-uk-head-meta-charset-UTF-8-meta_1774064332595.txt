<script>
/* ── tabs ── */
function sw(btn, id) {
  document.querySelectorAll('.dtab').forEach(b => b.classList.remove('on'));
  document.querySelectorAll('.view').forEach(v => v.classList.remove('on'));
  btn.classList.add('on');
  document.getElementById(id).classList.add('on');
  if (id === 'v3') startStream();
  else resetStream();
}

/* ── resources drawer ── */
let resOpen = false;
function toggleRes() {
  resOpen = !resOpen;
  document.getElementById('res-drawer').classList.toggle('open', resOpen);
  document.getElementById('res-drawer').setAttribute('aria-hidden', String(!resOpen));
  document.getElementById('res-toggle').setAttribute('aria-expanded', String(resOpen));
  document.getElementById('res-toggle').style.background = resOpen ? 'var(--p2)' : '';
  document.getElementById('res-toggle').style.borderColor = resOpen ? 'var(--border-strong)' : '';
  document.getElementById('res-toggle').style.color = resOpen ? 'var(--text-1)' : '';
}

/* ── textarea ── */
function grow(el) {
  el.style.height = 'auto';
  el.style.height = Math.min(el.scrollHeight, 120) + 'px';
}
function onKey(e, el) {
  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault();
    el.value = ''; el.style.height = '';
  }
}

/* ── feedback ── */
function dofb(btn, dir) {
  const row = btn.closest('.fbrow');
  row.querySelectorAll('.fb').forEach(b => b.classList.remove('up','dn'));
  btn.classList.add(dir);
  const lbl = row.querySelector('.fb-lbl');
  if (lbl) lbl.textContent = dir === 'up' ? 'Дякуємо! 🙏' : 'Зрозуміло, покращимо';
}

/* ── streaming ── */
const FULL =
  'Основні правила користування бібліотекою ХДАК:\n\n' +
  '1. Бережно ставитися до книг — не робити позначок\n' +
  '2. Повертати видання вчасно відповідно до терміну\n' +
  '3. Дотримуватися тиші в читальних залах\n' +
  '4. Пред\'являти читацький квиток на вимогу\n' +
  '5. Не виносити книги без належного оформлення';

let streaming = false;
function startStream() {
  if (streaming) return;
  streaming = true;
  resetStream();
  let i = 0;
  const stxt = document.getElementById('stxt');
  setTimeout(() => {
    document.getElementById('tgrp').style.display = 'none';
    document.getElementById('sgrp').style.display = '';
    function tick() {
      if (i >= FULL.length) return;
      const ch = FULL[i++];
      const cur = stxt.querySelector('.cur');
      if (ch === '\n') stxt.insertBefore(document.createElement('br'), cur);
      else stxt.insertBefore(document.createTextNode(ch), cur);
      setTimeout(tick, '.!?,'.includes(ch) ? 240 : Math.random() * 20 + 28);
    }
    tick();
  }, 900);
}
function resetStream() {
  streaming = false;
  const tg = document.getElementById('tgrp');
  const sg = document.getElementById('sgrp');
  const st = document.getElementById('stxt');
  if (tg) tg.style.display = '';
  if (sg) sg.style.display = 'none';
  if (st) st.innerHTML = '<span class="cur"></span>';
}
